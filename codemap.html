CREATE OR REPLACE FUNCTION `fn_get_random`(
  `c1` VARCHAR(64),
  `c2` VARCHAR(64),
  `c3` VARCHAR(64),
  `c4` VARCHAR(64),
  `c5` VARCHAR(64)
) RETURNS varchar(64)
COMMENT '5개 인자값 중 랜덤으로 하나를 선택'
BEGIN
  DECLARE res VARCHAR(64) DEFAULT '';
  DECLARE cvar INT(2) DEFAULT 0;
 
  SELECT
    FLOOR(1 + RAND() * (5 - 1 + 1))
  INTO cvar;
 
  IF cvar = 1 THEN
    SELECT c1 INTO res;
  ELSEIF cvar = 2 THEN
    SELECT c2 INTO res;
  ELSEIF cvar = 3 THEN
    SELECT c3 INTO res;
  ELSEIF cvar = 4 THEN
    SELECT c4 INTO res;
  ELSEIF cvar = 5 THEN
    SELECT c5 INTO res;
  END IF;
 
  RETURN res;
END //
